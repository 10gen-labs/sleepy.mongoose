 19 /**
 20  * Sleepy.Mongoose is the REST interface to Mongo. 
 21  *
 22  * Possible requests for Sleepy.Mongoose:
 23  *
 24  * POST requests:
 25  * <ul>
 26  *  <li>
 27  *   <pre>/_connect</pre>
 28  *   Required arguments:
 29  *   <ul>
 30  *    <li>server=<em>database server</em></li>
 31  *   </ul>
 32  *  </li>
 33  *  <li>
 34  *   <pre>/dbname/collection/_insert</pre>
 35  *   Required arguments:
 36  *   <ul>
 37  *    <li>docs=<an array of objs to insert></li>
 38  *   </ul>
 39  *   Returns:
 40  *   <pre>{"ok" : 1}</pre> 
 41  *  </li>
 42  *  <li>
 43  *   <pre>/dbname/collection/_remove</pre>
 44  *   Optional arguments:
 45  *   <ul>
 46  *    <li>criteria=<criteria for deletion></li>
 47  *   </ul>
 48  *   Returns:
 49  *   <pre>{"ok" : 1}</pre> 
 50  *  </li>
 51  *  <li>
 52  *   <pre>/dbname/collection/_update</pre>
 53  *   Required arguments:
 54  *   <ul>
 55  *    <li>criteria=<criteria for update></li>
 56  *    <li>newobj=<update content></li>
 57  *   </ul>
 58  *   Returns:
 59  *   <pre>{"ok" : 1}</pre> 
 60  *  </li>
 61  * </ul>
 62  *
 63  * GET requests:
 64  * <ul>
 65  *  <li>
 66  *   <pre>/_hello</pre>
 67  *   <p>Takes no arguments.  Returns a JSON object with "ok" and "msg" fields.</p>
 68  *  </li>
 69  *  <li>
 70  *   <pre>/dbname/collection/_find</pre>
 71  *   Optional arguments:
 72  *   <ul>
 73  *    <li>criteria=<search criteria></li>
 74  *    <li>fields=<fields to return></li>
 75  *    <li>skip=# to skip</li>
 76  *    <li>limit=# to return</li>
 77  *    <li>batch_size=# of results to return at a time, defaults to 15</li>
 78  *   </ul>
 79  *   Returns:
 80  *   <pre>{"ok" : 1, "results" : [docs], "id" : N}</pre> 
 81  *  </li>
 82  *  <li>
 83  *   <pre>/dbname/collection/_more</pre>
 84  *   Required arguments:
 85  *   <ul>
 86  *    <li>id=cursor id</li>
 87  *   </ul>
 88  *   Optional arguments:
 89  *   <ul>
 90  *    <li>batch_size=# of results to return at a time, defaults to 15</li>
 91  *   </ul>
 92  *   Returns:
 93  *   <pre>{"ok" : 1, "results" : [docs], "id" : N}</pre> 
 94  *  </li>
 95  * </ul>
 96  *
 97  * @name Sleepy.Mongoose 
 98  * @class 
 99  */
100 Sleepy.Mongoose = {};
101 
102 /**
103  * Constructs a new instance of Sleepy.Mongoose.
104  * @constructor
105  */
106 Sleepy.Mongoose.prototype = new Sleepy();
107 
108 Sleepy.Mongoose.prototype.constructor = function(host) {
109     this.server = host ? host : (Sleepy.Mongoose.server.host + ":" + Sleepy.Mongoose.server.port);
110 
111     if (connect !== false) {
112         this.connect();
113     }
114 
115     /**
116      * Connects to a db server.
117      *
118      * @param {String} [name] name to give connection
119      * @param {function} [callback] optional function to call when a response is
120      * received.
121      * @return undefined
122      * @throws Exception if callback is not a function
123      */
124     this.connect = function(name, callback) {
125         var args = "server="+this.server;
126         if (name && typeof name == "string") {
127             args += "&name="+name;
128         }
129 
130         this.post("/_connect", args, callback);
131     }
132 
133     this._doOp = function(op, action, db, collection, fields, callback) {
134         args = [];
135 
136         for (var key in fields) {
137             if (typeof fields[key] == "object") {
138                 args.push(key+"="+escape($.toJSON(this._pyhint(fields[key]))));
139             }
140             else {
141                 args.push(key+"="+fields[key]);
142             }
143         }
144 
145         this[action]("/"+db+"/"+collection+"/"+op, args.join("&"), callback);
146     }
147 
148     /**
149      * Query the database.
150      *
151      * Options can include:
152      * <dl>
153      *  <dt>criteria : <em>object</em></dt>
154      *  <dd>criteria for which to search</dd>
155      *  <dt>fields : <em>object</em></dt>
156      *  <dd>fields of results to return (_id is always returned)</dd>
157      *  <dt>limit : <em>N</em></dt>
158      *  <dd>number of results to return</dd>
159      *  <dt>skip : <em>N</em></dt>
160      *  <dd>number of results to skip</dd>
161      *  <dt>batch_size : N</dt>
162      *  <dd>number of results returned in the fields batch (defaults to 15)</dd>
163      * </dl>
164      *
165      * The callback receives an object of the form:
166      * <pre>
167      * {"results" : [], "id" : N, "ok" : 1}
168      * </pre>
169      * You must keep track of the id returned if you want to get more 
170      * results from this query. 
171      *
172      * @param {string} db database name
173      * @param {string} collection collection name
174      * @param {object} [options] optional options for query
175      * @param {function} [callback] callback to execute
176      * @return undefined
177      * @throws Exception if callback is not a function
178      */
179     this.find = function(db, collection, options, callback) {
180         this._doOp("_find", "get", db, collection, options, callback);
181     }
182 
183     /**
184      * Get more results from a cursor.
185      *
186      * Options must include:
187      * <dl>
188      *  <dt>id : <em>N</em></dt>
189      *  <dd>the id for this query (from the find)</dd>
190      * </dl>
191      *
192      * Options may include:
193      * <dl>
194      *  <dt>batch_size : <em>N</em></dt>
195      *  <dd>number of results returned in the fields batch (defaults to 15)</dd>
196      * </dl>
197      *
198      * The callback receives an object of the form:
199      * <pre>
200      * {"results" : [], "id" : N, "ok" : 1}
201      * </pre>
202      * You must keep track of the id returned if you want to get more 
203      * results from this query. 
204      *
205      * @param {string} db database name
206      * @param {string} collection collection name
207      * @param {object} options options for getting more results
208      * @param {function} [callback] callback to execute
209      * @return undefined
210      * @throws Exception if callback is not a function
211      */
212     this.more = function(db, collection, options, callback) {
213         this._doOp("_more", "get", db, collection, options, callback);
214     }
215 
216     /**
217      * Delete objects from a collection.
218      *
219      * Options may include:
220      * <dl>
221      *  <dt>criteria : <em>object</em></dt>
222      *  <dd>criteria for removing objects</dd>
223      * </dl>
224      *
225      * The callback receives an object of the form:
226      * <pre>
227      * {"ok" : 1}
228      * </pre>
229      *
230      * @param {string} db database name
231      * @param {string} collection collection name
232      * @param {object} [options] options for getting more results
233      * @param {function} [callback] callback to execute
234      * @return undefined
235      * @throws Exception if callback is not a function
236      */
237     this.remove = function(db, collection, options, callback) {
238         this._doOp("_remove", "post", db, collection, options, callback);
239     }
240 
241     /**
242      * Get more results from a cursor.
243      *
244      * Options must include:
245      * <dl>
246      *  <dt>criteria : <em>object</em></dt>
247      *  <dd>criteria to find objects to update</dd> 
248      *  <dt>newobj : <em>object</em></dt>
249      *  <dd>update to perform</dd>
250      * </dl>
251      *
252      * The callback receives an object of the form:
253      * <pre>
254      * {"ok" : 1}
255      * </pre>
256      *
257      * @param {string} db database name
258      * @param {string} collection collection name
259      * @param {object} options options for getting more results
260      * @param {function} [callback] callback to execute
261      * @return undefined
262      * @throws Exception if callback is not a function
263      */
264     this.update = function(db, collection, options, callback) {
265         this._doOp("_update", "post", db, collection, options, callback);
266     }
267 
268     /**
269      * Get more results from a cursor.
270      * Options must include:
271      * <dl>
272      *  <dt>docs : <em>array</em></dt>
273      *  <dd>an array of objects to insert</dd> 
274      * </dl>
275      *
276      * The callback receives an object of the form:
277      * <pre>
278      * {"ok" : 1}
279      * </pre>
280      *
281      * @param {string} db database name
282      * @param {string} collection collection name
283      * @param {object} options options for getting more results
284      * @param {function} [callback] callback to execute
285      * @return undefined
286      * @throws Exception if callback is not a function
287      */
288     this.insert = function(db, collection, options, callback) {
289         this._doOp("_insert", "post", db, collection, options, callback);
290     }
291 
292     /**
293      * Execute a database command.
294      *
295      * @param {string} db database name
296      * @param {object} obj the command to execute
297      * @param {function} [callback] optional function to call when a response is
298      * received.
299      * @return undefined
300      * @throws Exception if callback is not a function
301      */
302     this.command = function(db, obj, callback) {
303         uri = "/_cmd";
304         if (db && typeof db == "string") {
305             uri = "/" + db + uri;
306         }
307         this.post(uri, "obj="+escape($.toJSON(this._pyhint(obj))), callback);
308     }
309 };
310 
311 /**
312  * The default host an port to use if the constructor is not passed a server
313  * name.  This is an object with fields "host" (string) and "port" (number).
314  */
315 Sleepy.Mongoose.server = { host : "localhost", port : 27017 };
