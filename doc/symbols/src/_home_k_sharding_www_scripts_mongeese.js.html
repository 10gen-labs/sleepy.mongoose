 19 /**
 20  * This is the JavaScript sharding API
 21  *
 22  * It can be used by web interfaces to easily interact with the sharding 
 23  * backend.
 24  * @name Mongeese 
 25  * @class 
 26  */
 27 Mongeese = {};
 28 
 29 Mongeese.prototype = new Sleepy.Mongoose();
 30 
 31 /**
 32  * Sets the mongos to use.
 33  *
 34  * @param {String} server the mongos server to use
 35  * @param {function} [callback] optional function to call when a response is
 36  * received.
 37  * @return undefined
 38  * @throws Exception if callback is not a function
 39  */
 40 Mongeese.prototype.setMongos = function(callback) {
 41     this.connect("mongos", callback);
 42 };
 43 
 44 /**
 45  * Get the configuration server.
 46  *
 47  * @param {function} callback optional function to call when a response is
 48  * received.
 49  * @return undefined
 50  * @throws Exception if callback is not a function
 51  */
 52 Mongeese.prototype.getConfig = function(callback) {
 53     this.command("admin", {"netstat" : 1}, callback);
 54 };
 55 
 56 /**
 57  * Get a list of shards.
 58  *
 59  * @param {function} callback optional function to call when a response is
 60  * received.
 61  * @return undefined
 62  * @throws Exception if callback is not a function
 63  */
 64 Mongeese.prototype.getShards = function(callback) {
 65     this.command("admin", {"listshards" : 1}, callback);
 66 };

 69 /**
 70  * A Mongeese.Shard is an individual shard in the cluster.  It must be added to 
 71  * the cluster manually by calling "add".
 72  *
 73  * So, here's the workflow:
 74  * <ol>
 75  *  <li>
 76  *   You have an existing shard server (mongos) and config server, plus zero or
 77  *   more existing shards.  You have an instance of Mongoose associated with the
 78  *   mongos running.
 79  *  </li>
 80  *  <li>
 81  *   You start up a new shard.  No one knows that this shard even exists yet.
 82  *   Say it is listening for connections at localhost:12345.
 83  *  </li>
 84  *  <li>
 85  *   You create a Mongoose.Shard instance for this shard, passing it the 
 86  *   Mongoose instance and the address of the shard ("localhost:12345").
 87  *  </li>
 88  *  <li>
 89  *   Finally, we let mongos and the config server know about this new shard by 
 90  *   calling the add method on this shard.
 91  *  </li>
 92  * </ol>
 93  *
 94  * @constructor
 95  */
 96 Mongeese.Shard = function(mongeese, server) {
 97 
 98     this.connection = mongeese;
 99     this.server = server;
100 
101     /**
102      * Add this shard.
103      *
104      * The callback will receive a response of the form:
105      * <pre>
106      * { "added" : "localhost:10000", "ok" : 1 }
107      * </pre>
108      *
109      * If you do not specify local and the shard is on localhost, you will get:
110      * <pre>
111      * {
112      *     "ok" : 0,
113      *     "errmsg" : "can't use localhost as a shard since all shards need to communicate.  allowLocal to override for testing"
114      * }
115      * </pre>
116      *
117      * @param {boolean} [local] optional parameter specifying if the shard is on localhost
118      * @param {function} [callback] optional function to call when a response is
119      * received.
120      * @return undefined
121      * @throws Exception if callback is not a function
122      */
123     this.add = function(local, callback) {
124         local = local ? true : false;
125         this.connection.command("admin", {"addshard" : this.server, "allowLocal" : local}, callback);
126     };
127 
128     /**
129      * Not yet implemented (in the db).
130      */
131     this.remove = function() {
132         // TODO
133     }
134 
135     /**
136      * Checks the status of this server, including uptime, memory usage, and 
137      * activity.
138      *
139      * The callback will receive a response of the form:
140      * <pre>
141      * {
142      *     "uptime" : 52358,
143      *     "globalLock" : {
144      *          "totalTime" : 52358205700,
145      *          "lockTime" : 1455600,
146      *          "ratio" : 0.000027800799904034907
147      *     },
148      *     "mem" : {
149      *          "resident" : 22,
150      *          "virtual" : 145,
151      *          "supported" : true,
152      *          "mapped" : 80
153      *     },
154      *     "connections" : {
155      *          "current" : 2,
156      *          "available" : 19998
157      *     },
158      *     "extra_info" : {
159      *          "note" : "fields vary by platform",
160      *          "heap_usage_bytes" : 1874632,
161      *          "page_faults" : 1
162      *     },
163      *     "indexCounters" : {
164      *          "btree" : {
165      *              "accesses" : 1,
166      *              "hits" : 1,
167      *              "misses" : 0,
168      *              "resets" : 0,
169      *              "missRatio" : 0
170      *          }
171      *     },
172      *     "opcounters" : {
173      *          "insert" : 3,
174      *          "query" : 15,
175      *          "update" : 0,
176      *          "delete" : 0,
177      *          "getmore" : 0
178      *     },
179      *     "ok" : 1
180      * }
181      * </pre>
182      *
183      * @param {function} callback optional function to call when a response is
184      * received.
185      * @return undefined
186      * @throws Exception if callback is not a function
187      */
188     this.status = function(callback) {
189         this.connection.get("/_status", "server="+this.server, callback);
190     };
191 
192     /**
193      * Get a list of databases on this shard.
194      *
195      * @param {function} callback optional function to call when a response is
196      * received.
197      * @return undefined
198      * @throws Exception if callback is not a function
199      */
200     this.getDatabases = function(callback) {
201         this.connection.get("/_dbs", "server="+this.server, callback);
202     };
203 
204 }

206 /**
207  * A sharded database.
208  *
209  * @constructor
210  */
211 Mongeese.Database = function(mongeese, db) {
212     this.connection = mongeese;
213     this.db = db + "";
214     
215     /**
216      * Enables sharding on a database.
217      *
218      * @param {String} db the database to enable sharding on
219      * @param {function} callback optional function to call when a response is
220      * received.
221      * @return undefined
222      * @throws Exception if callback is not a function
223      */
224     this.shard = function(callback) {
225         this.connection.command("admin", {"enablesharding" : this.db}, callback);
226     };
227 
228     /**
229      * Move the database to a different shard.
230      *
231      * @param {String} to the server to move the database to
232      * @param {function} callback optional function to call when a response is
233      * received.
234      * @return undefined
235      * @throws Exception if callback is not a function
236      */
237     this.move = function(to, callback) {
238         this.connection.command("admin", {"moveprimary" : this.db, "to" : to}, callback);
239     }
240 };

242 /**
243  * A sharded collection.  Used to manipulate individual chunks (pieces of a 
244  * collection).
245  *
246  * @constructor
247  */
248 Mongeese.Collection = function(db, collection) {
249     this.connection = db.connection;
250     this.ns = db.db + "." + collection;
251 
252     /**
253      * Enables sharding on this collection.  The collection's database must have 
254      * sharding enabled before it can be enabled on the collection.
255      *
256      * @param {object} key the field or fields on which to shard, e.g., {"x" : 1}
257      * @param {boolean} [unique] if the shard key should be unique
258      * @param {function} [callback] optional function to call when a response is
259      * received.
260      * @return undefined
261      * @throws Exception if callback is not a function
262      */
263     this.shard = function(key, unique, callback) {
264         unique = unique ? true : false;
265         this.connection.command("admin", {"shardcollection" : this.ns, "key" : key, "unique" : unique}, callback);
266     };
267 
268     /**
269      * Split this collection into two chunks on a single shard.  Use 
270      * Mongeese.Collection.move to move one of the chunks to a new shard.
271      *
272      * @param {object} criteria must contain "find" OR "middle" field which 
273      * gives the split criteria.
274      * @param {function} [callback] optional function to call when a response is
275      * received.
276      * @return undefined
277      * @throws Exception if criteria doesn't contain "find" or "middle" fields 
278      * or callback is not a function
279      */
280     this.split = function(criteria, callback) {
281 
282         var cmd = {"shard" : this.ns};
283 
284         if (criteria.find != null) {
285             cmd.find = criteria.find;
286         }
287         else if (criteria.middle != null) {
288             cmd.middle = criteria.middle;
289         }
290         else {
291             throw "no find or middle object given";
292         }
293 
294         this.connection.command("admin", cmd, callback);
295     };
296 
297     /**
298      * Move a chunk of this collection to a new shard.  Use the criteria 
299      * parameter to find the chunk to move.
300      *
301      * @param {object} criteria must contain "find" field which matches at least
302      * one document in the chunk to move.
303      * @param {string} to the shard to which to move the chunk, e.g., 
304      * "localhost:27019"
305      * @param {function} [callback] optional function to call when a response is
306      * received.
307      * @return undefined
308      * @throws Exception if criteria doesn't contain a "find" field
309      * or callback is not a function
310      */
311     this.move = function(criteria, to, callback) {
312         var cmd = {"movechunk" : this.ns};
313 
314         if (criteria.find != null) {
315             cmd.find = criteria.find;
316         }
317         else {
318             throw "no find object given";
319         }
320         cmd.to = to;
321 
322         this.connection.command("admin", cmd, callback);
323     };
324 
325     /**
326      * Get the version of sharding being used.
327      *
328      * @param {function} callback optional function to call when a response is
329      * received.
330      * @return undefined
331      * @throws Exception if callback is not a function
332      */
333     this.version = function(callback) {
334         this.connection.command("admin", {"getShardVersion" : this.ns}, callback);
335     };
336 }
