 19 /**
 20  * Sleepy is the basic REST JavaScript interface that Mongoose is build on top 
 21  * of.
 22  *
 23  * @constructor
 24  */
 25 Sleepy = function(host, connect) {
 26     this.httpd = Sleepy.httpd.host + ":" + Sleepy.httpd.port;
 27 
 28     /**
 29      * This turns an object into an array of pairs so that Python won't mess up
 30      * the ordering.
 31      */
 32     this._pyhint = function(data) {
 33         var kv = [];
 34         for(var key in data) {
 35             kv.push({"key" : key, "value" : data[key]});
 36         }
 37         return {'$pyhint' : kv};
 38     }
 39 
 40 
 41     /**
 42      * This creates an AJAX request.  It sends an asynchronous request to
 43      * the server and calls an optional callback when a response is received, if
 44      * no error occured.
 45      *
 46      * @param {String} uri the address to send the request
 47      * @param {String} data key/value pairs to send
 48      * @param {callback} function a function to call when the response is 
 49      * received.
 50      * @return undefined
 51      * @throws Exception if callback is not a function
 52      */
 53     this._ajax = function(action, uri, data, callback) {
 54         if (callback === undefined) {
 55             callback = Sleepy.defaultCallback;
 56         }
 57         else if (typeof callback != "function") {
 58             throw "callback is not a function";
 59         }
 60 
 61         $.ajax({
 62                 "type" : action,
 63                 "url" : this.httpd+uri,
 64                 "data" : data,
 65                 "dataType" : "json",
 66                 "success" : function(msg) {
 67                     var result = Sleepy.handleError(msg);
 68                     if (result.ok == 0) {
 69                         return;
 70                     }
 71 
 72                     callback(msg);
 73                 },
 74                 // TODO: better error handling
 75                 "error" : function(request, status, err) {
 76                     Sleepy.handleError(err);
 77                 }
 78             });
 79     }
 80 
 81     /**
 82      * This creates a POST AJAX request.  It sends an asynchronous request to
 83      * the server and calls an optional callback when a response is received, if
 84      * no error occured.
 85      *
 86      * @param {String} uri the address to send the request
 87      * @param {String} data key/value pairs to send
 88      * @param {callback} function a function to call when the response is 
 89      * received.
 90      * @return undefined
 91      * @throws Exception if callback is not a function
 92      */
 93     this.post = function(uri, data, callback) {
 94         this._ajax("POST", uri, data, callback);
 95     };
 96 
 97     /**
 98      * This creates a GET AJAX request.  It sends an asynchronous request to
 99      * the server and calls an optional callback when a response is received, if
100      * no error occured.
101      *
102      * @param {String} uri the address to send the request
103      * @param {String} data key/value pairs to send
104      * @param {callback} function a function to call when the response is 
105      * received.
106      * @return undefined
107      * @throws Exception if callback is not a function
108      */
109     this.get = function(uri, data, callback) {
110         this._ajax("GET", uri, data, callback);
111     };
112 };
113 
114 Sleepy.httpd = { host : "http://localhost", port : 27080 };
115 
116 Sleepy.defaultCallback = function(msg) { return; };
117 
118 /**
119  * Handles errors in database responses.
120  * 
121  * @returns an object.  If the "ok" field is 0, there will be a "msg" field 
122  * that will describe the error that occured.  If the command succeeded (or
123  * appeared to) the "ok" field will be 1.
124  * @type Object
125  */
126 Sleepy.handleError = function(msg) {
127     if (!msg) {
128         return {"ok" : 0, "msg" : "Something went very wrong. You'll probably need to restart the server."};
129     }
130     else if (!msg['ok']) {
131         return {"ok" : 0, "msg" : msg['msg'] ? msg['msg'] : msg['errmsg']};
132     }
133     return {"ok" : 1};
134 };
