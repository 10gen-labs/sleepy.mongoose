1 Sleepy.Mongoose.Util = function() {
  2 
  3     /**
  4      * Number of seconds/time unit
  5      * [sec/year, sec/month, sec/day, sec/hour, sec/minute]
  6      */
  7     var time_vals = [31104000, 2592000, 86400, 3600, 60];
  8 
  9     /**
 10      * Pretty names for time units
 11      */
 12     var time_units = ['year', 'month', 'day', 'hour', 'minute'];
 13 
 14     /**
 15      * Pretty names for byte units
 16      */
 17     var byte_units = ['bytes', 'kB', 'MB', 'GB', 'TB', 'PB'];
 18 
 19     /**
 20      * Turn number of seconds into a "pretty" time representation: years, 
 21      * months, days, etc.
 22      *
 23      * @param {Number} time number of seconds
 24      * @return the prettified time
 25      * @type String
 26      */
 27     get_time_format = function(time) {
 28         str = "";
 29 
 30         remaining = time;
 31         for (var i in time_vals) {
 32             if ((num = Math.floor(remaining/time_vals[i])) > 0) {
 33                 str += num + " " + time_units[i] + (num == 1 ? "" : "s") + " ";
 34                 remaining = remaining - (time_vals[i] * num);
 35             }
 36         }
 37 
 38         if (remaining > 0) {
 39             str += remaining + " seconds";
 40         }
 41 
 42         return str;
 43     };
 44 
 45     /**
 46      * Get the "pretty' size.
 47      * Uses the fact that sizes are measured base-10 as a hack.
 48      *
 49      * @param {Number} bytes size in bytes
 50      * @returns prettified size
 51      * @type String
 52      */
 53     var get_byte_format = function(bytes) {
 54         var size = Math.floor(((bytes+"").length-1)/3);
 55         bytes = bytes/Math.pow(1000, size);
 56         return bytes + " " + byte_units[size];
 57     }
 58 
 59 };
